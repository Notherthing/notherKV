cmake_minimum_required(VERSION 3.10)

#project name
PROJECT(test)


#c11 standard
SET(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)     # Require C++17 support

SET(CMAKE_CXX_FLAGS "-static-libstdc++ -static-libgcc -O3 -ggdb ") 

find_package(Threads)

#src lists
SET(
SRC_LIST_CLI 
tcpCliTest.cpp 
)

#head file path
INCLUDE_DIRECTORIES(
/include/
)

#lib file path
LINK_DIRECTORIES(
/lib
)

#set extern libraries
SET(LIBRARIES
libmiux_base.a
)

# set output binary path
SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin)

#add executable file
ADD_EXECUTABLE(cliTest ${SRC_LIST_CLI})

#add link library
TARGET_LINK_LIBRARIES(cliTest ${LIBRARIES} ${CMAKE_THREAD_LIBS_INIT})
